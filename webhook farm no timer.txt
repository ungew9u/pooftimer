local http_request = request or syn.request or fluxus.request;

function comma_value(amount) -- credits: https://devforum.roblox.com/t/how-would-i-make-a-large-number-have-commas/384427/2
    local formatted = amount;
    while true do  
        formatted, k = string.gsub(formatted, "^(-?%d+)(%d%d%d)", "%1,%2");
        if (k == 0) then
            break;
        end;
    end;
    return formatted;
end;
